const denoms=[10000,5000,1000,500,100,50,10,5,1];

const tbody=document.getElementById('tbody');

const totalEl=document.getElementById('total');

function makeRow(d){const tr=document.createElement('tr');tr.innerHTML=`<td>${d}</td><td><input type="number" value="0" min="0"></td><td class="subtotal">0</td>`;tbody.appendChild(tr);tr.querySelector('input').addEventListener('input',update);}

function update(){let total=0;for(const tr of tbody.children){const d=Number(tr.children[0].textContent);const c=Number(tr.children[1].querySelector('input').value);const sub=d*c;tr.children[2].textContent=sub.toLocaleString();total+=sub;}totalEl.textContent=total.toLocaleString();}

denoms.forEach(makeRow);update();
